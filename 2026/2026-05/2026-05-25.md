## カバリングインデックス

### 概要

- これを読み解く：[11.9. インデックスオンリースキャンとカバリングインデックス](https://www.postgresql.jp/document/18/html/indexes-index-only-scans.html)
  - ポスグレではテーブルの本体が格納されている領域をヒープといい、インデックスはヒープとは別で格納されている
  - つまり、インデックスを使うということは、まずインデックスの領域からデータを取得し、それを元にヒープにアクセスして実際のデータを取得するという二段階の処理が必要になる
  - また、インデックス内では類似データは近くに配置されるが、それらがヒープではどこに配置されているかわからない
  - つまり、インデックスを使ったとしてもヒープへのアクセスが遅い可能性がある
  - これを解決するために、インデックスのみを使ってクエリに応答するインデックスオンリースキャンがPostgreSQLでは可能
  - 条件
    - インデックスオンリースキャンをサポートしているインデックスであること
      - B-Treeインデックスではサポートされているが、他のインデックスは一部のみサポートまたは未サポート
    - クエリがインデックスに指定されている列のみ使っていること
    - 対象のデータが可視性マップで可視であること
      - このページは全トランザクションから見える、という情報が記録されているもの
      - [66.4. 可視性マップ](https://www.postgresql.jp/document/18/html/storage-vm.html)
      - もし可視ではなかったら、それが可視かどうかを調べるためにヒープへのアクセスが必要で、性能は向上しない
  - 頻繁に更新される値には有効ではないが、時系列データなどで基本的に凍結されたデータには有効

- インデックスサイズは通常のインデックスに比べて大きくなるため以下のデメリットはある
  - 更新が遅くなる
  - ディスク容量をより多く使用する
- そのためなんでもかんでも詰め込めばいいというわけではなく、本当によく使うパターンにする、INCLUDEに入れる列は最小限にする、定期的に見直し不要であれば削除するなどが必要

-- ① テーブル作成
DROP TABLE IF EXISTS sensor_readings;

CREATE TABLE sensor_readings (
    id              BIGSERIAL       PRIMARY KEY,
    sensor_id       INTEGER         NOT NULL,
    temperature     NUMERIC(5,2)    NOT NULL,
    humidity        NUMERIC(5,2)    NOT NULL,
    pressure        NUMERIC(7,2)    NOT NULL,
    location        VARCHAR(50)     NOT NULL,
    battery_level   SMALLINT        NOT NULL,
    signal_strength SMALLINT        NOT NULL,
    recorded_at     TIMESTAMP       NOT NULL
);

-- ② 100万件 一括INSERT（50万件 × 2バッチ）
DO $$
DECLARE
    batch_size  INT := 500000;
    total       INT := 1000000;  -- ここを変えるだけ
    i           INT := 0;
    locations   TEXT[] := ARRAY['屋上','屋内','駐車場','玄関','倉庫','工場フロア','屋外フェンス'];
BEGIN
    WHILE i < total LOOP
        INSERT INTO sensor_readings (sensor_id, temperature, humidity, pressure, location, battery_level, signal_strength, recorded_at)
        SELECT
            1 + (gs % 1000),
            ROUND((-10 + (gs % 500) * 0.1)::NUMERIC, 2),
            ROUND((30 + (gs % 7001) * 0.01)::NUMERIC, 2),
            ROUND((960 + (gs % 8001) * 0.005)::NUMERIC, 2),
            locations[1 + (gs % array_length(locations, 1))],
            (gs % 101),
            -120 + (gs % 121),
            TIMESTAMP '2020-01-01' + ((gs % 1576800) || ' minutes')::INTERVAL
        FROM generate_series(i + 1, i + batch_size) AS gs;

        i := i + batch_size;
        RAISE NOTICE '% 件 INSERT完了', i;
    END LOOP;
END $$;

DO $$
DECLARE
    batch_size      INT := 500000;
    additional      INT := 1000000;  -- ← 追加したい件数をここで指定
    start_gs        INT;
    i               INT;
    locations       TEXT[] := ARRAY['屋上','屋内','駐車場','玄関','倉庫','工場フロア','屋外フェンス'];
BEGIN
    -- 現在の最大gsを取得して続きから始める
    SELECT COALESCE(MAX(id), 0) INTO start_gs FROM sensor_readings;
    i := start_gs;

    WHILE i < start_gs + additional LOOP
        INSERT INTO sensor_readings (sensor_id, temperature, humidity, pressure, location, battery_level, signal_strength, recorded_at)
        SELECT
            1 + (gs % 1000),
            ROUND((-10 + (gs % 500) * 0.1)::NUMERIC, 2),
            ROUND((30 + (gs % 7001) * 0.01)::NUMERIC, 2),
            ROUND((960 + (gs % 8001) * 0.005)::NUMERIC, 2),
            locations[1 + (gs % array_length(locations, 1))],
            (gs % 101),
            -120 + (gs % 121),
            TIMESTAMP '2020-01-01' + ((gs % 1576800) || ' minutes')::INTERVAL
        FROM generate_series(i + 1, i + batch_size) AS gs;

        i := i + batch_size;
        RAISE NOTICE '合計約 % 件', i;
    END LOOP;
END $$;


-- ③ 件数確認
SELECT COUNT(*) AS total_rows FROM sensor_readings;
