既存のsqlをDSQL互換に変換する

```
npx aurora-dsql-prisma transform prisma/migrations/20250822055143_add_users_table/migration.sql -o migration.sql
```

```sql
-- ### オリジナル ###
-- CreateTable
CREATE TABLE "public"."users" (
    "id" SERIAL NOT NULL,
    "name" VARCHAR(100) NOT NULL,
    "email" VARCHAR(255) NOT NULL,
    "created_at" TIMESTAMP(6) DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT "users_pkey" PRIMARY KEY ("id")
);

-- CreateIndex
CREATE UNIQUE INDEX "users_email_key" ON "public"."users"("email");

-- ### 変換後 ###
CREATE TABLE "public"."users" (
  "id" BIGINT NOT NULL GENERATED BY DEFAULT AS IDENTITY ( CACHE 1 ),
  "name" VARCHAR(100) NOT NULL,
  "email" VARCHAR(255) NOT NULL,
  "created_at" TIMESTAMP(6) DEFAULT CURRENT_TIMESTAMP,
  CONSTRAINT "users_pkey" PRIMARY KEY ("id")
);

CREATE UNIQUE INDEX ASYNC "users_email_key" ON "public"."users"("email");
```

現在の`schema.prisma`ファイルからDSQL互換のマイグレーションファイルを作成する

```
npx prisma migrate diff \
    --from-empty \
    --to-schema prisma/schema.prisma \
    --script | npx aurora-dsql-prisma transform > migration.sql
```

↑が何をしているかというと普通にPrismaのコマンドで`schema.prisma`からマイグレーションファイルを作成した後、このライブラリを使って変換している

下記コマンドでも実行可能

※これを実行する場合、`relationMode=prisma`の記載が必要

```
npx aurora-dsql-prisma migrate prisma/schema.prisma -o prisma/migrations/001_init/migration.sql
```

作成されるSQL

```sql
CREATE SCHEMA IF NOT EXISTS "public";

CREATE TABLE "users" (
  "id" BIGINT NOT NULL GENERATED BY DEFAULT AS IDENTITY ( CACHE 1 ),
  "first_name" VARCHAR(100) NOT NULL,
  "email" VARCHAR(255) NOT NULL,
  "created_at" TIMESTAMP(6) DEFAULT CURRENT_TIMESTAMP,
  "age" INTEGER,
  CONSTRAINT "users_pkey" PRIMARY KEY ("id")
);

CREATE TABLE "posts" (
  "id" BIGINT NOT NULL GENERATED BY DEFAULT AS IDENTITY ( CACHE 1 ),
  "authorId" INTEGER NOT NULL,
  "created_at" TIMESTAMP(6) DEFAULT CURRENT_TIMESTAMP,
  CONSTRAINT "posts_pkey" PRIMARY KEY ("id")
);

CREATE UNIQUE INDEX ASYNC "users_email_key" ON "users"("email");

CREATE INDEX ASYNC "posts_created_at_idx" ON "posts"("created_at");
```

SERIAL型は`GENERATED BY DEFAULT AS IDENTITY`で作成されるが、AWSとしての推奨は、書き込み量の多いテーブルでは単調増加する整数よりランダムな主キーを推奨。そのためテーブルの要件によってはUUIDにしても良い。

```
model users {
  id         String    @id @default(dbgenerated("gen_random_uuid()")) @db.Uuid
  first_name String    @db.VarChar(100)
  email      String    @unique @db.VarChar(255)
  created_at DateTime? @default(now()) @db.Timestamp(6)
  age        Int?

  posts posts[]
}
```
